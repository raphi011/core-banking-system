% PDF-only preamble (pandoc --include-in-header for the LaTeX/tectonic build).
%
% The body font (Latin Modern) lacks a few technical symbols the prose uses, so
% map them onto LaTeX equivalents — otherwise they are silently dropped. These
% mappings fire only in normal prose: inside verbatim/code blocks the characters
% keep their literal catcodes and are drawn by the monospace font instead (which
% is why the ASCII diagrams need a monofont with box-drawing coverage, set on the
% pandoc command line).
% Fit wide ASCII diagrams. pandoc emits plain `verbatim`, so redefine it as a
% natural-width boxed verbatim (fancyvrb's BVerbatim) wrapped in adjustbox: any
% block wider than the text column is scaled down to fit, preserving the diagram's
% alignment; normal-width code is left at full size.
\usepackage{fancyvrb}
\usepackage{adjustbox}
\usepackage{etoolbox}
\DefineVerbatimEnvironment{verbatim}{BVerbatim}{}
\BeforeBeginEnvironment{verbatim}{\par\noindent\begin{adjustbox}{max width=\linewidth}}
\AfterEndEnvironment{verbatim}{\end{adjustbox}\par}

\usepackage{newunicodechar}
\usepackage{textcomp}
\newunicodechar{→}{\textrightarrow}   % U+2192 rightwards arrow
\newunicodechar{−}{\textminus}        % U+2212 minus sign
\newunicodechar{×}{\texttimes}        % U+00D7 multiplication sign
\newunicodechar{€}{\texteuro}         % U+20AC euro sign
\newunicodechar{≠}{\ensuremath{\neq}} % U+2260 not equal to
